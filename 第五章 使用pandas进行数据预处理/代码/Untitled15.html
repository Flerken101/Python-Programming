#定义3sigma原则识别异常值函数
import pandas as pd
detail = pd.read_csv('G:/《Python数据分析与应用》/Python数据分析与应用/第5章/任务程序/data/detail.csv',index_col= 0,encoding='gbk')

def outRange(Ser1):
    boolInd = (Ser1.mean()-3*Ser1.std()>Ser1)|(Ser1.mean()+3*Ser1.std()<Ser1) #判断所选列每个值是否是异常值（共有209个True,即209个异常值)
    boolInd.to_csv('G:/《Python数据分析与应用》/第五章/boolInd.csv',encoding='gbk',header='True')

    '''因为boolInd前面以detail_id作为索引，但是会有两行记录为同一个detail_id，
    且这两行记录的counts值不同，有的counts可能为异常值(True)，有的可能不是异常值(False)
    index中只留下异常值的唯一索引（从0开始，不再使用detail_id），209个，并储存在一个一维数组中'''
    print(Ser1.shape[0])
    index = np.arange(Ser1.shape[0])[boolInd] 
    print(index)
    print(len(index)) #209
    
    #根据索引将每个异常值从原数据中提取出来，具体到counts列
    outrange = Ser1.iloc[index] 
    return outrange


outlier = outRange(detail['counts'])
outlier.to_csv('G:/《Python数据分析与应用》/第五章/outlier.csv',encoding='gbk',header='True')

print('异常值的最小值为：',outlier.min())

def FeatureEquals(df):
    dfEquals=pd.DataFrame([],columns=df.columns,index=df.columns)
    for i in df.columns:
        for j in df.columns:
            dfEquals.loc[i,j] = df.loc[:,i].equals(df.loc[:,j])
    return dfEquals
detEquals=FeatureEquals(detail) #应用上述函数

dupCol = []

    for i in range(k+1,lenDet):
        if detEquals.iloc[k,i] & (detEquals.columns[i] not in dupCol):
            dupCol.append(detEquals.columns[i])
#删除重复列
