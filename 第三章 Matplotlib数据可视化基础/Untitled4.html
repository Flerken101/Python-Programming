#pyplot中的基础绘图语法
import numpy as np
import matplotlib.pyplot as plt
# %matplotlib inline 表示在行中显示图片，在命令行运行报错
data = np.arange(0,1.1,0.1)
plt.title('lines') #添加标题
plt.xlabel('x')
plt.xlabel('y')#添加两轴的名称
plt.xlim((0,1))
plt.ylim((0,1))#确定两轴的范围
plt.xticks(np.arange(0,1.2,0.2))
plt.yticks(np.arange(0,1.2,0.2))#规定两轴的刻度
plt.plot(data,data**2)
plt.plot(data,data**4) #添加曲线
plt.legend(['y=x^2','y=x^4'])
plt.savefig('G:/《Python数据分析与应用》学习代码/第三章/y=x^2.png')
plt.show()

ax1 = p1 .add_subplot(2,1,1)#创建一个两行一列的子图，并开始绘制第一幅

plt.xlabel('x')
plt.ylabel('y')

plt.ylim((0,1))

plt.yticks(np.arange(0,1.2,0.2))

plt.plot(rad,rad**4)

plt.plot(rad,np.cos(rad))

#绘制2000-2017年各季度国民生产总值散点图
import numpy as np
import matplotlib.pyplot as plt
plt.rcParams['font.sans-serif'] = 'SimHei'
plt.rcParams['axes.unicode_minus'] = False
data = np.load('G:/《Python数据分析与应用》/Python数据分析与应用/第3章/任务程序/data/国民经济核算季度数据.npz')
name = data['columns'] #提取其中的columns数组，视为数据的标签
values = data['values']  #提取其中的values数组，视为数据的存在位置
plt.figure(figsize=(8,7)) #设置画布
plt.title('2000-2017年各季度国民生产总值散点图') #添加图表标题
plt.xlabel('年份')
plt.ylabel('生产总值（亿元）')
plt.ylim((0,225000)) #设置y轴范围
plt.xticks(range(0,70,4),values[range(0,70,4),1],rotation=45)
plt.scatter(values[:,0],values[:,2],maker='。') #绘制散点图
plt.savefig("G:/《Python数据分析与应用》/第三章/2000-2017年各季度国民生产总值散点图.png")
plt.show()
